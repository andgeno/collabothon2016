contract singlePerson {
    address owner;
    function singlePerson() { owner = msg.sender; }
    function kill() { if (msg.sender == owner) selfdestruct(owner); }
    
    uint256 money = 0;
    function pay() public payable {
        money = msg.value;
    }
    
    function check() public constant returns (bool){
        return money != 0;
    }
}
